# 记忆面包 闲时计算架构设计方案

## 设计目标

1. **最小化内存占用** - 大部分时间只加载必需模型
2. **闲时批量计算** - 在用户空闲时处理积压任务
3. **串行加载模型** - 同时最多加载 1-2 个模型，避免内存爆炸
4. **智能调度** - 根据任务优先级和系统资源动态调度

---

### 1. 闲时检测器（Idle Detector）

#### 检测指标

```python
class IdleDetector:
    """系统闲时检测器"""

    def __init__(self):
        self.idle_threshold = 300  # 5 分钟
        self.cpu_threshold = 20    # CPU < 20%
        self.memory_threshold = 70  # 内存 < 70%

    def is_idle(self) -> bool:
        """判断系统是否处于闲时"""
        # 1. 检查用户空闲时间
        idle_time = self.get_system_idle_time()
        if idle_time < self.idle_threshold:
            return False

        # 2. 检查 CPU 使用率
        cpu_usage = psutil.cpu_percent(interval=1)
        if cpu_usage > self.cpu_threshold:
            return False

        # 3. 检查内存使用率
        memory = psutil.virtual_memory()
        if memory.percent > self.memory_threshold:
            return False

        # 4. 检查是否在充电（笔记本）
        if not self.is_on_power():
            return False

        return True

    def get_system_idle_time(self) -> int:
        """获取系统空闲时间（秒）"""
        # macOS
        if platform.system() == "Darwin":
            output = subprocess.check_output(
                ["ioreg", "-c", "IOHIDSystem"]
            )
            for line in output.decode().split('\n'):
                if 'HIDIdleTime' in line:
                    ns = int(line.split('=')[1].strip())
                    return ns // 1_000_000_000

        # Windows
        elif platform.system() == "Windows":
            import ctypes
            class LASTINPUTINFO(ctypes.Structure):
                _fields_ = [
                    ('cbSize', ctypes.c_uint),
                    ('dwTime', ctypes.c_uint),
                ]

            lii = LASTINPUTINFO()
            lii.cbSize = ctypes.sizeof(LASTINPUTINFO)
            ctypes.windll.user32.GetLastInputInfo(ctypes.byref(lii))
            millis = ctypes.windll.kernel32.GetTickCount() - lii.dwTime
            return millis // 1000

        return 0

    def is_on_power(self) -> bool:
        """检查是否在充电"""
        battery = psutil.sensors_battery()
        if battery is None:
            return True  # 台式机
        return battery.power_plugged
```

### 2. 任务调度器（Task Scheduler）

#### 任务队列设计

```python
from enum import Enum
from dataclasses import dataclass
from typing import List
import asyncio

class TaskPriority(Enum):
    """任务优先级"""
    REALTIME = 0    # 实时（OCR）
    ON_DEMAND = 1   # 按需（LLM）
    HIGH = 2        # 高优先级（用户主动触发）
    NORMAL = 3      # 普通（Embedding）
    LOW = 4         # 低优先级（VLM）

class TaskType(Enum):
    """任务类型"""
    OCR = "ocr"
    EMBEDDING = "embedding"
    ASR = "asr"
    VLM = "vlm"
    LLM = "llm"

@dataclass
class Task:
    """计算任务"""
    id: int
    type: TaskType
    priority: TaskPriority
    data: dict
    created_at: float

class TaskScheduler:
    """闲时任务调度器"""

    def __init__(self, idle_detector: IdleDetector):
        self.idle_detector = idle_detector
        self.task_queue: List[Task] = []
        self.running = False

    def add_task(self, task: Task):
        """添加任务到队列"""
        self.task_queue.append(task)
        # 按优先级排序
        self.task_queue.sort(key=lambda t: (t.priority.value, t.created_at))

    async def start(self):
        """启动调度器"""
        self.running = True

        while self.running:
            # 每 30 秒检查一次
            await asyncio.sleep(30)

            # 检查是否闲时
            if not self.idle_detector.is_idle():
                continue

            # 处理任务队列
            await self.process_queue()

    async def process_queue(self):
        """处理任务队列（闲时批量计算）"""
        if not self.task_queue:
            return

        logger.info(f"进入闲时计算模式，待处理任务: {len(self.task_queue)}")

        # 按任务类型分组
        tasks_by_type = {}
        for task in self.task_queue:
            if task.type not in tasks_by_type:
                tasks_by_type[task.type] = []
            tasks_by_type[task.type].append(task)

        # 串行处理每种类型的任务
        for task_type, tasks in tasks_by_type.items():
            # 检查是否还在闲时
            if not self.idle_detector.is_idle():
                logger.info("系统不再空闲，暂停闲时计算")
                break

            # 加载对应模型并批量处理
            await self.process_batch(task_type, tasks)

    async def process_batch(self, task_type: TaskType, tasks: List[Task]):
        """批量处理同类型任务"""
        logger.info(f"开始处理 {task_type.value} 任务，数量: {len(tasks)}")

        # 1. 加载模型
        model = await ModelManager.load(task_type)

        try:
            # 2. 批量处理
            if task_type == TaskType.EMBEDDING:
                await self.batch_embedding(model, tasks)
            elif task_type == TaskType.ASR:
                await self.batch_asr(model, tasks)
            elif task_type == TaskType.VLM:
                await self.batch_vlm(model, tasks)

            # 3. 从队列中移除已完成的任务
            for task in tasks:
                self.task_queue.remove(task)

        finally:
            # 4. 卸载模型
            await ModelManager.unload(task_type)
            logger.info(f"{task_type.value} 模型已卸载")

    async def batch_embedding(self, model, tasks: List[Task]):
        """批量向量化"""
        texts = [task.data['text'] for task in tasks]

        # 分批处理（每批 32 条）
        batch_size = 32
        for i in range(0, len(texts), batch_size):
            batch = texts[i:i+batch_size]
            vectors = model.encode(batch)

            # 写入数据库
            for j, vector in enumerate(vectors):
                task = tasks[i+j]
                await self.save_vector(task.data['capture_id'], vector)

            # 检查是否还在闲时
            if not self.idle_detector.is_idle():
                logger.info("系统不再空闲，暂停批量向量化")
                break

    async def batch_asr(self, model, tasks: List[Task]):
        """批量语音识别"""
        for task in tasks:
            audio_path = task.data['audio_path']
            result = model.transcribe(audio_path)
            await self.save_asr_result(task.data['capture_id'], result)

            # 检查是否还在闲时
            if not self.idle_detector.is_idle():
                break

    async def batch_vlm(self, model, tasks: List[Task]):
        """批量图像理解"""
        for task in tasks:
            image_path = task.data['image_path']
            prompt = task.data['prompt']
            result = model.generate(image_path, prompt)
            await self.save_vlm_result(task.data['capture_id'], result)

            # 检查是否还在闲时
            if not self.idle_detector.is_idle():
                break
```

### 3. 模型管理器（Model Manager）

#### 串行加载策略

```python
class ModelManager:
    """模型生命周期管理器（串行加载）"""

    _instance = None
    _loaded_models = {}  # 当前已加载的模型
    _lock = asyncio.Lock()  # 全局锁，确保串行加载

    @classmethod
    async def load(cls, task_type: TaskType):
        """加载模型（串行，同时最多 1 个模型）"""
        async with cls._lock:
            # 如果已加载，直接返回
            if task_type in cls._loaded_models:
                return cls._loaded_models[task_type]

            # 卸载其他模型（确保同时只有 1 个模型）
            await cls.unload_all()

            # 加载新模型
            logger.info(f"加载模型: {task_type.value}")
            model = await cls._load_model(task_type)
            cls._loaded_models[task_type] = model

            return model

    @classmethod
    async def _load_model(cls, task_type: TaskType):
        """实际加载模型"""
        if task_type == TaskType.EMBEDDING:
            from embedding.model import EmbeddingModel
            return EmbeddingModel.create_quantized()  # 600MB

        elif task_type == TaskType.ASR:
            from asr.model import AsrModel
            return AsrModel.create_tiny()  # 80MB

        elif task_type == TaskType.VLM:
            # 通过 Ollama 加载
            import subprocess
            subprocess.run(["ollama", "run", "minicpm-v:8b-2.6-q2_K"])
            return "minicpm-v-q2"  # 3GB

        elif task_type == TaskType.LLM:
            # 通过 Ollama 加载
            import subprocess
            subprocess.run(["ollama", "run", "qwen2.5:3b-instruct-q4_K_M"])
            return "qwen2.5-3b"  # 2.5GB

    @classmethod
    async def unload(cls, task_type: TaskType):
        """卸载指定模型"""
        if task_type in cls._loaded_models:
            logger.info(f"卸载模型: {task_type.value}")
            del cls._loaded_models[task_type]

            # 强制 GC
            import gc
            gc.collect()

    @classmethod
    async def unload_all(cls):
        """卸载所有模型"""
        for task_type in list(cls._loaded_models.keys()):
            await cls.unload(task_type)
```
